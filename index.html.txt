        // ==========================================
        // 1. CẤU HÌNH GOOGLE DRIVE API & XÁC THỰC

        const CLIENT_ID = '525283300446-kr7c6g9ho426fo0s0n139qcngv0o8htb.apps.googleusercontent.com'; 
        const SCOPES = 'https://www.googleapis.com/auth/drive.appdata'; 
        
        let tokenClient;
        let gDriveAccessToken = null;
        let progressFileId = null;
        let cachedProgress = {};
        let syncTimeout = null;

        window.onload = function() {
            // Khởi tạo Google Identity Services
            tokenClient = google.accounts.oauth2.initTokenClient({
                client_id: CLIENT_ID,
                scope: SCOPES,
                callback: async (tokenResponse) => {
                    if (tokenResponse && tokenResponse.access_token) {
                        gDriveAccessToken = tokenResponse.access_token;
                        document.getElementById('btn-login-google').classList.add('hidden');
                        document.getElementById('btn-logout-google').classList.remove('hidden');
                        document.getElementById('sync-status').textContent = '🔄 Đang tải dữ liệu...';
                        document.getElementById('sync-status').classList.replace('text-gray-500', 'text-blue-500');
                        await initDriveStorage();
                    }
                },
            });
        };

        function loginGoogle() {
            tokenClient.requestAccessToken();
        }

        function logoutGoogle() {
            gDriveAccessToken = null;
            cachedProgress = {};
            progressFileId = null;
            document.getElementById('btn-login-google').classList.remove('hidden');
            document.getElementById('btn-logout-google').classList.add('hidden');
            document.getElementById('sync-status').textContent = '🚫 Đã đăng xuất';
            document.getElementById('sync-status').classList.replace('text-green-600', 'text-gray-500');
            document.getElementById('sync-status').classList.replace('text-blue-500', 'text-gray-500');
        }

        async function initDriveStorage() {
            try {
                // 1. Tìm file tiến trình trong thư mục ẩn appDataFolder
                const listRes = await fetch("https://www.googleapis.com/drive/v3/files?spaces=appDataFolder&q=name='reading_progress.json'", {
                    headers: { Authorization: `Bearer ${gDriveAccessToken}` }
                });
                const listData = await listRes.json();
                
                if (listData.files && listData.files.length > 0) {
                    // File đã tồn tại, tiến hành tải nội dung
                    progressFileId = listData.files[0].id;
                    const fileRes = await fetch(`https://www.googleapis.com/drive/v3/files/${progressFileId}?alt=media`, {
                        headers: { Authorization: `Bearer ${gDriveAccessToken}` }
                    });
                    
                    try {
                        cachedProgress = await fileRes.json();
                    } catch(e) { cachedProgress = {}; }

                } else {
                    // Chưa có file, tạo file mới tinh
                    const createRes = await fetch("https://www.googleapis.com/drive/v3/files", {
                        method: "POST",
                        headers: {
                            Authorization: `Bearer ${gDriveAccessToken}`,
                            "Content-Type": "application/json"
                        },
                        body: JSON.stringify({
                            name: "reading_progress.json",
                            parents: ["appDataFolder"]
                        })
                    });
                    const createData = await createRes.json();
                    progressFileId = createData.id;
                    cachedProgress = {};
                }

                document.getElementById('sync-status').textContent = '✅ Đã kết nối Drive';
                document.getElementById('sync-status').classList.replace('text-blue-500', 'text-green-600');

                // Nếu đang mở sách dở, tự động chuyển đến trang vừa tải về
                if (currentBookId && cachedProgress[currentBookId]) {
                    restoreProgress(cachedProgress[currentBookId]);
                }

            } catch (error) {
                console.error("Lỗi khởi tạo Drive:", error);
                document.getElementById('sync-status').textContent = '⚠️ Lỗi đồng bộ Drive';
                document.getElementById('sync-status').classList.replace('text-blue-500', 'text-red-500');
            }
        }

        function saveProgressToDrive(bookId, progressValue) {
            // Lưu vào RAM ngay lập tức
            cachedProgress[bookId] = progressValue;

            if (!gDriveAccessToken || !progressFileId) return; // Nếu chưa đăng nhập thì không làm gì thêm

            const statusEl = document.getElementById('sync-status');
            statusEl.textContent = '⏳ Đang chờ lưu...';
            statusEl.classList.replace('text-green-600', 'text-blue-500');

            // Kỹ thuật Debounce: Gom nhiều lần lật trang liên tiếp thành 1 lần lưu để tránh khóa API Google
            clearTimeout(syncTimeout);
            syncTimeout = setTimeout(async () => {
                try {
                    statusEl.textContent = '🔄 Đang tải lên Drive...';
                    await fetch(`https://www.googleapis.com/upload/drive/v3/files/${progressFileId}?uploadType=media`, {
                        method: "PATCH",
                        headers: {
                            Authorization: `Bearer ${gDriveAccessToken}`,
                            "Content-Type": "application/json"
                        },
                        body: JSON.stringify(cachedProgress)
                    });
                    statusEl.textContent = '✅ Đã lưu tiến độ';
                    statusEl.classList.replace('text-blue-500', 'text-green-600');
                    setTimeout(() => { 
                        if(statusEl.textContent === '✅ Đã lưu tiến độ') statusEl.textContent = '✅ Đã kết nối Drive'; 
                    }, 3000);
                } catch (e) {
                    statusEl.textContent = '⚠️ Lỗi khi lưu';
                    statusEl.classList.replace('text-blue-500', 'text-red-500');
                }
            }, 2500); // Đợi 2.5s sau lần lật trang cuối cùng mới lưu
        }

        function restoreProgress(savedValue) {
            if (!savedValue) return;
            if (activeFileType === 'pdf') {
                let p = parseInt(savedValue);
                if (p > 0 && p <= totalPdfPages) {
                    currentPdfPage = p;
                    renderPdfPage(currentPdfPage);
                    showToast(`Đã khôi phục trang ${currentPdfPage} từ Google Drive`);
                }
            } else if (activeFileType === 'epub' && epubRendition) {
                epubRendition.display(savedValue).then(() => {
                    showToast("Đã khôi phục tiến trình EPUB từ Google Drive");
                });
            }
        }

        // ==========================================
        // 2. LOGIC TRÌNH ĐỌC SÁCH
        // ==========================================
        pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.11.174/pdf.worker.min.js';

        let currentPdf = null, currentPdfPage = 1, totalPdfPages = 0, currentPdfScale = 1.2;
        let currentEpub = null, epubRendition = null;
        let activeFileType = null; 
        let currentBookId = null;

        const fileInput = document.getElementById('file-input');
        const epubViewer = document.getElementById('epub-viewer');
        const pdfContainer = document.getElementById('pdf-container');
        const pdfCanvas = document.getElementById('pdf-canvas');
        const controls = document.getElementById('controls');
        const emptyState = document.getElementById('empty-state');
        const prevBtn = document.getElementById('prev-btn');
        const nextBtn = document.getElementById('next-btn');
        const pageInfo = document.getElementById('page-info');
        const loadingScreen = document.getElementById('loading');
        const loadingText = document.getElementById('loading-text');
        const toast = document.getElementById('toast');
        const toastMessage = document.getElementById('toast-message');

        function showToast(message) {
            toastMessage.textContent = message;
            toast.classList.remove('opacity-0', 'pointer-events-none');
            setTimeout(() => toast.classList.add('opacity-0', 'pointer-events-none'), 3000);
        }

        function showLoading(text = "Đang xử lý, vui lòng đợi...") {
            loadingText.textContent = text;
            loadingScreen.classList.remove('hidden');
        }

        function hideLoading() {
            loadingScreen.classList.add('hidden');
        }

        function resetViewer() {
            emptyState.classList.add('hidden');
            if (currentEpub) { currentEpub.destroy(); currentEpub = null; epubRendition = null; }
            if (currentPdf) currentPdf = null;
            epubViewer.innerHTML = '';
            pdfCanvas.getContext('2d').clearRect(0, 0, pdfCanvas.width, pdfCanvas.height);
        }

        fileInput.addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;

            currentBookId = "local_" + file.name.replace(/[^a-zA-Z0-9]/g, '_') + '_' + file.size; 
            const filename = file.name.toLowerCase();
            showLoading("Đang đọc file của bạn...");

            const reader = new FileReader();
            if (filename.endsWith('.pdf')) {
                reader.onload = (evt) => loadPDF(new Uint8Array(evt.target.result));
            } else if (filename.endsWith('.epub')) {
                reader.onload = (evt) => loadEPUB(evt.target.result);
            } else {
                hideLoading();
                showToast("Định dạng không hỗ trợ! Vui lòng chọn PDF/EPUB");
                fileInput.value = '';
                return;
            }
            reader.readAsArrayBuffer(file);
        });

        // ==========================
        // XỬ LÝ FILE PDF
        // ==========================
        async function loadPDF(typedarray) {
            resetViewer();
            activeFileType = 'pdf';
            epubViewer.classList.add('hidden');
            pdfContainer.classList.remove('hidden');
            controls.classList.remove('hidden');

            if (window.innerWidth < 600) currentPdfScale = 0.8;
            else if (window.innerWidth < 1000) currentPdfScale = 1.0;
            else currentPdfScale = 1.3;

            try {
                currentPdf = await pdfjsLib.getDocument(typedarray).promise;
                totalPdfPages = currentPdf.numPages;
                currentPdfPage = 1;

                // Kiểm tra tiến trình đã lưu
                if (cachedProgress[currentBookId]) {
                    currentPdfPage = parseInt(cachedProgress[currentBookId]);
                    if (currentPdfPage < 1 || currentPdfPage > totalPdfPages) currentPdfPage = 1;
                }

                renderPdfPage(currentPdfPage);
            } catch (error) {
                hideLoading();
                showToast("Không thể đọc file PDF này.");
                console.error(error);
            }
        }

        function renderPdfPage(num) {
            showLoading(`Đang nạp trang ${num}...`);
            currentPdf.getPage(num).then(page => {
                const viewport = page.getViewport({ scale: currentPdfScale });
                pdfCanvas.height = viewport.height;
                pdfCanvas.width = viewport.width;

                page.render({ canvasContext: pdfCanvas.getContext('2d'), viewport: viewport }).promise.then(() => {
                    pageInfo.textContent = `${currentPdfPage} / ${totalPdfPages}`;
                    pdfContainer.scrollTop = 0; 
                    
                    if (currentBookId) saveProgressToDrive(currentBookId, currentPdfPage);
                    hideLoading();
                });
            });
        }

        // ==========================
        // XỬ LÝ FILE EPUB
        // ==========================
        function loadEPUB(arrayBuffer) {
            resetViewer();
            activeFileType = 'epub';
            pdfContainer.classList.add('hidden');
            epubViewer.classList.remove('hidden');
            controls.classList.remove('hidden');
            pageInfo.textContent = 'EPUB';

            try {
                currentEpub = ePub(arrayBuffer);
                const epubInner = document.createElement('div');
                epubInner.className = 'epub-container';
                epubViewer.appendChild(epubInner);

                epubRendition = currentEpub.renderTo(epubInner, {
                    width: "100%", height: "100%", spread: "none"
                });
                
                epubRendition.themes.default({
                    'body': { 'background': 'transparent !important', 'padding': '0 10px !important' },
                    'p': { 'font-size': '1.1rem !important', 'line-height': '1.6 !important', 'color': '#374151 !important' }
                });

                const savedCfi = cachedProgress[currentBookId];
                const displayPromise = savedCfi ? epubRendition.display(savedCfi) : epubRendition.display();

                displayPromise.then(() => { hideLoading(); });

                epubRendition.on("relocated", function(location) {
                    pageInfo.textContent = 'Trang sách'; 
                    if (currentBookId && location && location.start) {
                        saveProgressToDrive(currentBookId, location.start.cfi);
                    }
                });

            } catch (error) {
                hideLoading();
                showToast("Lỗi khi mở file EPUB.");
            }
        }

        // ĐIỀU HƯỚNG SÁCH

        function openLinkModal() { document.getElementById('link-modal').classList.remove('hidden'); }
        function closeLinkModal() { document.getElementById('link-modal').classList.add('hidden'); }

        function processCloudLink(url) {
            const gdriveMatch = url.match(/\/file\/d\/([a-zA-Z0-9_-]+)/);
            if (gdriveMatch) return `https://drive.google.com/uc?export=download&id=${gdriveMatch[1]}`;
            if (url.includes('dropbox.com') && url.includes('dl=0')) return url.replace('dl=0', 'dl=1');
            return url;
        }

        async function loadFromLink() {
            let url = document.getElementById('link-input').value.trim();
            const typeOverride = document.getElementById('link-type').value;
            if (!url) { showToast("Vui lòng nhập đường dẫn URL!"); return; }

            closeLinkModal();
            showLoading("Đang tải sách từ mạng...");
            currentBookId = "link_" + btoa(url).replace(/[^a-zA-Z0-9]/g, '').substring(0, 30); // Mã hóa URL làm ID

            try {
                const processedUrl = processCloudLink(url);
                const proxies = [
                    'https://api.codetabs.com/v1/proxy?quest=' + encodeURIComponent(processedUrl),
                    'https://corsproxy.io/?' + encodeURIComponent(processedUrl),
                    'https://api.allorigins.win/raw?url=' + encodeURIComponent(processedUrl)
                ];

                let response = null;
                for (const proxyUrl of proxies) {
                    try {
                        const res = await fetch(proxyUrl);
                        if (res.ok) { response = res; break; }
                    } catch (e) {}
                }
                
                if (!response) throw new Error("Không thể tải file qua các proxy dự phòng.");

                const arrayBuffer = await response.arrayBuffer();
                let fileType = typeOverride;

                if (fileType === 'auto') {
                    const header = new Uint8Array(arrayBuffer).subarray(0, 4).reduce((acc, val) => acc + val.toString(16), "");
                    if (header.startsWith("25504446")) fileType = 'pdf';
                    else if (header.startsWith("504b0304")) fileType = 'epub';
                    else throw new Error("Không thể nhận diện định dạng, vui lòng chọn thủ công.");
                }

                if (fileType === 'pdf') loadPDF(new Uint8Array(arrayBuffer));
                else if (fileType === 'epub') loadEPUB(arrayBuffer);

            } catch (err) {
                hideLoading();
                showToast("Lỗi: " + err.message);
            }
        }
